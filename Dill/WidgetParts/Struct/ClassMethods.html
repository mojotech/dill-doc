def list(name, selector, options = {}, &block)
  child = widget(name, selector, Dill::List) do
    item options[:item_selector], options[:item_class] || ListItem
  end

  class_eval <<-WIDGET
    def #{name}
      widget(:#{name}).value
    end
  WIDGET

  child.class_eval(&block) if block_given?

  child
end
