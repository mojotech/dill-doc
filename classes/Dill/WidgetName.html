def to_class(scope = Object)
  const = scope.const_get(@canonical)

  raise TypeError, "`#{@canonical}' is not a widget in this scope"          unless const < Widget

  const
rescue NameError
  raise Missing, "couldn't find `#{@name}' widget in this scope"
end
